import cv2
import numpy as np
import face_recognition as fr

video_capture = cv2.VideoCapture(0)

image_tata = fr.load_image_file('D:\\face_detection\\img\\tata.jpg')
image_face_tata = fr.face_encodings(image_tata)[0]

image_principle = fr.load_image_file('D:\\face_detection\\img\\Dr.kulkarni sir.jpg')
image_face_principle = fr.face_encodings(image_principle)[0]

image_hod = fr.load_image_file('D:\\face_detection\\img\\Hon.Patange sir.jpg')
image_face_hod = fr.face_encodings(image_hod)[0]

image_shubham = fr.load_image_file('D:\\face_detection\\img\\IMG-20220106-WA0011.jpg')
image_face_shubbham = fr.face_encodings(image_shubham)[0]

image_pratham = fr.load_image_file('D:\\face_detection\\img\\Pratahm.png')
image_face_pratham = fr.face_encodings(image_pratham)[0]

image_sanket = fr.load_image_file('D:\\face_detection\\img\\Sanket.jpg')
image_face_sanket = fr.face_encodings(image_sanket)[0]

image_vaishnavi = fr.load_image_file('D:\\face_detection\\img\\vaishnavi.jpg')
image_face_vaishnavi = fr.face_encodings(image_vaishnavi)[0]

image_prathamesh = fr.load_image_file('D:\\face_detection\\img\\Prathamesh.jpg')
image_face_prathamesh = fr.face_encodings(image_prathamesh)[0]

known_face_encoding = [image_face_tata,
                      image_face_principle,
                      image_face_hod,
                      image_face_shubbham,
                      image_face_pratham,
                      image_face_sanket,
                      image_face_vaishnavi,
                      image_face_prathamesh,
                      ]
known_face_names = ["Tata",
                   "Dr.kulkarni sir",
                   "Hon.Patange sir","Shubham","Pratham","Sanket","Vaishnavi","Prathamesh"]

while True:
    ret, frame = video_capture.read()
    
    rgb_frame = frame[:, :, ::-1]

    fc_locations = fr.face_locations(rgb_frame)
    fc_encodings = fr.face_encodings(rgb_frame, fc_locations)

    for(top, right, bottom, left), face_encoding in zip(fc_locations, fc_encodings):
        mathces = fr.compare_faces(known_face_encoding, face_encoding)

        name = "unknown"
    
        fc_distances = fr.face_distance(known_face_encoding, face_encoding)

        match_index = np.argmin(fc_distances)

        if mathces[match_index]:
            name = known_face_names[match_index]

        cv2.rectangle(frame,(left, top),(right, bottom),2)

        cv2.rectangle(frame, (left, bottom - 35), (right, bottom), (0,0,225),cv2.FILLED)

        font = cv2.FONT_HERSHEY_COMPLEX

        cv2.putText(frame, name, (left +6, bottom -6), font, 1.0, (255, 255, 255),1)

    cv2.imshow('face detection system', frame)

    if cv2.waitKey(1)& 0xFF == ord('q'):
        break

video_capture.release()
cv2.destroyAllWindows()
